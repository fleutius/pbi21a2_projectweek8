Create User

For at lave nye brugere databasen bruger vi koden:

CREATE USER iud_user1@localhost IDENTIFIED BY 'pass1';
CREATE USER iud_user2@localhost IDENTIFIED BY 'pass2';
CREATE USER iud_user3@localhost IDENTIFIED BY 'pass3';
CREATE USER i_user@localhost IDENTIFIED BY 'pass4';

Create Role

CREATE ROLE

i_user;
